lxc添加映射目录脚本

命令 
nano lxc.sh
配置块 
#!/bin/bash

# 检查是否以 root 用户运行
if [ "$(id -u)" -ne 0 ]; then
    echo "请以 root 用户运行此脚本。"
    exit 1
fi

# 询问需要挂载的 LXC 容器 ID 或名称
read -p "请输入要挂载目录的 LXC 容器 ID：" CONTAINER_ID

# 检查容器配置文件是否存在
CONF_FILE="/etc/pve/lxc/${CONTAINER_ID}.conf"
if [ ! -f "$CONF_FILE" ]; then
    echo "容器配置文件 $CONF_FILE 不存在，请检查容器 ID 是否正确。"
    exit 1
fi

# 查找已有的 mp 数量，确定下一个可用的 mp 编号
LAST_MP=$(grep -E '^mp[0-9]+:' "$CONF_FILE" | tail -n1 | sed -E 's/^mp([0-9]+):.*/\1/')
if [ -z "$LAST_MP" ]; then
    NEXT_MP=0
else
    NEXT_MP=$((LAST_MP + 1))
fi

# 询问主机目录
read -p "请输入主机上要挂载的目录（例如 /data）：" HOST_DIR

# 检查主机目录是否存在，不存在则创建
if [ ! -d "$HOST_DIR" ]; then
    echo "主机目录 $HOST_DIR 不存在，是否创建？(y/n)"
    read CREATE_DIR
    if [ "$CREATE_DIR" = "y" ] || [ "$CREATE_DIR" = "Y" ]; then
        mkdir -p "$HOST_DIR"
        echo "已创建主机目录 $HOST_DIR。"
    else
        echo "主机目录 $HOST_DIR 不存在，且未创建。退出。"
        exit 1
    fi
fi

# 询问容器内的挂载点目录
read -p "请输入容器内的挂载目录（例如 /var/data）：" CONTAINER_DIR

# 确认信息
echo "即将在容器 $CONTAINER_ID 中将主机目录 $HOST_DIR 挂载到容器内的 $CONTAINER_DIR。"
read -p "确认继续吗？(y/n) " CONFIRM
if [ "$CONFIRM" != "y" ] && [ "$CONFIRM" != "Y" ]; then
    echo "操作已取消。"
    exit 0
fi

# 调整主机目录的权限
echo "正在调整主机目录的权限..."
chown -R 100000:100000 "$HOST_DIR"
chmod -R 755 "$HOST_DIR"

# 添加挂载点到容器配置文件
MP_ENTRY="mp${NEXT_MP}: ${HOST_DIR},mp=${CONTAINER_DIR},backup=0"
echo "$MP_ENTRY" >> "$CONF_FILE"
echo "已将以下挂载点添加到容器配置文件 $CONF_FILE："
echo "$MP_ENTRY"

# 重启容器以使配置生效
echo "是否现在重启容器 $CONTAINER_ID 以使配置生效？(y/n)"
read RESTART
if [ "$RESTART" = "y" ] || [ "$RESTART" = "Y" ]; then
    echo "正在重启容器 $CONTAINER_ID..."
    pct stop "$CONTAINER_ID"
    pct start "$CONTAINER_ID"
    echo "容器 $CONTAINER_ID 已重启。"
else
    echo "请稍后手动重启容器 $CONTAINER_ID 以使配置生效。"
fi

echo "操作完成。"
